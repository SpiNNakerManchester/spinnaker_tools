362SPIN1_INLINE void spin1_store_barrier(void)
363{
364#ifdef __GNUC__
365 // GCC only really has one kind of barrier, and this is how to do it
366 asm volatile("" : : : "cc", "memory");
367#else
368 __force_stores();
369#endif

377#else
378 __memory_changed();

392extern uint spin1_irq_disable(void);
393#else
394SPIN1_INLINE uint spin1_irq_disable(void)
395{
396 uint old_val, new_val;
397
398#ifdef __GNUC__
399 asm volatile (
400 "mrs %[old_val], cpsr \n\
401 orr %[new_val], %[old_val], #0x80 \n\
402 msr cpsr_c, %[new_val] \n"
403 : [old_val] "=r" (old_val), [new_val] "=r" (new_val)
404 :
405 : );
406#else
407 __asm { mrs old_val, cpsr }
408 __asm { orr new_val, old_val, 0x80 }
409 __asm { msr cpsr_c, new_val }
410#endif
411
412 spin1_memory_barrier();
413 return old_val;
414}
415#endif
